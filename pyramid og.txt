#include <stdio.h>

int main()
{
    int i,n,j;
    printf("enter a numbr");
    scanf("%d",&n);
    
     for(i=1;i<=n;i++)
    {
        for(j=n;j>=1;j--)
        
        {
            if(i>=j)
            {
       printf("* ");
            }
            else
            {
            printf(" ");
            }
        }     
        printf("\n");
    }
    
      
    return 0;
}
